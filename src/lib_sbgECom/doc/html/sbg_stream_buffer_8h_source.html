206

211  //

213  //
214  switch (origin)
215  {
216  case SB_SEEK_SET:
217  pHandle->pCurrentPtr = pHandle->pBufferPtr + offset;
218  break;
219  case SB_SEEK_CUR_INC:
220  pHandle->pCurrentPtr += offset;

222  case SB_SEEK_CUR_DEC:
223  pHandle->pCurrentPtr -= offset;

225  case SB_SEEK_END:
226  pHandle->pCurrentPtr = pHandle->pBufferPtr + (pHandle->bufferSize - offset);
227  break;

230  }
